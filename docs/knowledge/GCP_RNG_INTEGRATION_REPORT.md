    async def correlate_event(

```python
# whitening.py — Whitney-based whitening equivalent to GCP 2.0 processing
import numpy as np
from scipy import stats

def whiten_bit_stream(bits: np.ndarray, block_size: int = 200) -> np.ndarray:
    """
    Apply GCP-compatible whitening to raw bit stream.
    1. XOR successive bits (Von Neumann debiasing)
    2. Block-sum to Gaussian via CLT (block_size bits per sample)
    3. Normalize to z-score
    """
    # Von Neumann debiasing: XOR pairs
    pairs = bits.reshape(-1, 2)
    xored = pairs[:, 0] ^ pairs[:, 1]
    valid = xored[pairs[:, 0] != pairs[:, 1]]

    # Block-sum (Central Limit Theorem Gaussianization)
    n_blocks = len(valid) // block_size
    blocks = valid[:n_blocks * block_size].reshape(n_blocks, block_size)
    block_sums = blocks.sum(axis=1)

    # Normalize: expected mean = block_size/2, std = sqrt(block_size/4)
    expected_mean = block_size / 2
    expected_std = np.sqrt(block_size / 4)
    z_scores = (block_sums - expected_mean) / expected_std

    return z_scores

def compute_netvar(egg_z_scores: list[np.ndarray]) -> float:
    """
    Compute Network Coherence (netvar) = chi-square aggregated across Eggs.
    High netvar = network is more coherent than expected by chance.
    """
    all_z = np.concatenate(egg_z_scores)
    chi_sq = np.sum(all_z ** 2)
    dof = len(all_z)
    # Normalize to per-dof netvar
    return chi_sq / dof  # Expected ~1.0 under H0; > 1.0 indicates coherence

def stouffers_z(z_scores: list[float]) -> float:
    """Aggregate individual z-scores via Stouffer’s method."""
    return np.sum(z_scores) / np.sqrt(len(z_scores))
```
